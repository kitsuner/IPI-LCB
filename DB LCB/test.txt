select * from ingredient i INNER JOIN ss_ss_groupe_ingredients  ss on 
ss.id_ss_ss_grp_ingredient = i.id_ss_ss_groupe_ingredient 

where ss.lib_ss_ss_grp_ingredient= "sels";

select * from ss_groupe_ingredients ss INNER JOIN groupe_ingredients grp 
on ss.id_grp_ingredient=grp.id_grp_ingredient 
where grp.lib_grp_ingredient = "Boissons";

select * from ss_ss_groupe_ingredients ssss INNER JOIN ss_groupe_ingredients ss 
on ssss.id_ss_grp_ingredient=ss.id_ss_grp_ingredient 

where ss.lib_ss_grp_ingredient = "Viandes crues";
